308 void ZJointSpectraDataManager::zp_calculateCalibrationQualityData(bool saveToCalibration,
309  qint64 calibrationId,
310  int factorCount,
311  qreal summSquareAverageConcentrationDispersion) const
312 {
313  ZCalibrationQualityData qualityData;
314  if(calibrationId < 0)
315  {
316  return;
317  }
318 
319  QString chemElement = zv_calibrationRepository->zp_chemElementForCalibrationId(calibrationId);
320  qint64 calibrationChemElementId = zv_spectrumArrayRepository->zp_chemElementIdForName(zv_currentArrayIndex, chemElement);
321 
322  if(calibrationChemElementId < 0)
323  {
324  emit zg_calibrationQualityData(saveToCalibration, calibrationId, qualityData);
325  return;
326  }
327 
328  int checkedSpectrumCount = 0;
329  const ZAbstractSpectrum* spectrum;
330  QStringList calibrationConcentrationList = zv_calibrationConcentrationMap.value(calibrationId);
331  if(calibrationConcentrationList.isEmpty() || calibrationConcentrationList.count() != zv_spectrumArrayRepository->zp_spectrumCount(zv_currentArrayIndex))
332  {
333  emit zg_calibrationQualityData(saveToCalibration, calibrationId, qualityData);
334  return;
335  }
336 
337  qreal calibrationConcentration;
338  qreal chemConcentration;
339  qreal summSquareConcentrationDispersion = 0.0;
340  bool ok;
341 
342  for(int s = 0; s < zv_spectrumArrayRepository->zp_spectrumCount(zv_currentArrayIndex); s++)
343  {
344  spectrum = zv_spectrumArrayRepository->zp_spectrum(zv_currentArrayIndex, s);
345  if(spectrum == 0 || !spectrum->zp_isSpectrumChecked())
346  {
347  continue;
348  }
349 
350  checkedSpectrumCount++;
351  // chemConc
352  chemConcentration = spectrum->zp_concentrationValue(calibrationChemElementId);
353  // calibr conc
354  calibrationConcentration = calibrationConcentrationList.at(s).toDouble(&ok);
355  if(!ok)
356  {
357  emit zg_calibrationQualityData(saveToCalibration, calibrationId, qualityData);
358  return;
359  }
360 
361  summSquareConcentrationDispersion += pow(chemConcentration - calibrationConcentration, 2.0);
362  }
363 
364  if(checkedSpectrumCount <= 1)
365  {
366  emit zg_calibrationQualityData(saveToCalibration, calibrationId, qualityData);
367  return;
368  }
369 
370  // dispersion sigma square
371  qreal squareSigma = summSquareConcentrationDispersion / checkedSpectrumCount;
372  qualityData.squareSigma = QString::number(squareSigma);
373 
374  // standard deviation
375  qreal standardDeviation = sqrt(summSquareConcentrationDispersion / (checkedSpectrumCount - 1));
376  qualityData.standardDeviation = QString::number(standardDeviation);
377 
378  // R2 determination
379  if(summSquareAverageConcentrationDispersion == 0)
380  {
381  //
382  qualityData.determination = "Error";
383  qualityData.adj_determination = "Error";
384  emit zg_calibrationQualityData(saveToCalibration, calibrationId, qualityData);
385  return;
386  }
387 
388  qreal determination = 1 - (summSquareConcentrationDispersion / summSquareAverageConcentrationDispersion);
389  qualityData.determination = QString::number(determination);
390 
391  // R2adj determination
392  qreal denominator = checkedSpectrumCount - factorCount - 1;
393  if(denominator == 0)
394  {
395  qualityData.adj_determination = "Error";
396  emit zg_calibrationQualityData(saveToCalibration, calibrationId, qualityData);
397  return;
398  }
399 
400  qreal adjDetermination = 1 - (1 - determination)*( (checkedSpectrumCount - 1)/(denominator) );
401  qualityData.adj_determination = QString::number(adjDetermination);
402  emit zg_calibrationQualityData(saveToCalibration, calibrationId, qualityData);
403

827 bool ZJointSpectraDataManager::zp_calculateConcentrationResidualList(qint64 calibrationId,
828  QList<qreal>&residualDispersionList) const
829 {
830  residualDispersionList.clear();
831  QString chemElementName = zv_calibrationRepository->zp_chemElementForCalibrationId(calibrationId);
832  qint64 chemElementId = zv_spectrumArrayRepository->zp_chemElementIdForName(zv_currentArrayIndex, chemElementName);
833 
834  if(chemElementId < 0)
835  {
836  return false;
837  }
838 
839  qreal concentration;
840  const ZAbstractSpectrum* spectrum;
841  bool res;
842  qreal residual;
843  qreal averageResidual = 0.0;
844  int checkedSpectrumCount = 0;
845  for(int s = 0; s < zv_spectrumArrayRepository->zp_spectrumCount(zv_currentArrayIndex); s++)
846  {
847  spectrum = zv_spectrumArrayRepository->zp_spectrum(zv_currentArrayIndex, s);
848  if(!spectrum)
849  {
850  return false;
851  }
852 
853  if(!spectrum->zp_isSpectrumChecked())
854  {
855  continue;
856  }
857  checkedSpectrumCount ++;
858  // calculate calibration concentration value
859  res = zv_calibrationRepository->zp_calculateConcentrationForId(calibrationId, spectrum, concentration);
860  if(res)
861  {
862  // get chem concentration
863  //calc residual and put it in residualList
864  residual = spectrum->zp_concentrationValue(chemElementId) - concentration;
865  averageResidual += residual;
866  residualDispersionList.append(residual);
867  }
868  else
869  {
870  return false;
871  }
872  }
873 
874  if(checkedSpectrumCount < 1)
875  {
876  return false;
877  }
878 
879  averageResidual /= static_cast<qreal>(checkedSpectrumCount);
880 
881  for(int r = 0; r < residualDispersionList.count(); r++)
882  {
883  residualDispersionList[r] = residualDispersionList.at(r) - averageResidual;
884  }
885  return true;
886 }
