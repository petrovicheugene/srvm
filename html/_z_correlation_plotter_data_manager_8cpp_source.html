234 
235  // get kind of chart

237  if(!vData.isValid() || vData.isNull() || !vData.canConvert<int>())

239  // TODO Error msg
240  return;
241  }
242 
243  ChartDataKind chartDataKind = static_cast<ChartDataKind>(vData.toInt());
244 
245  // getting chart data

248  qreal maxX = 0.0;

252 
253  if(chartDataKind == CDK_TERM)
254  {
255  zv_defaultItem->zp_setMedianaLineVisible(false);
256  if(!zh_getTermToConcentrationData(chartPointMap, maxX, minX, maxY, minY))
257  {
258  // TODO ERROR msg

262  zh_recalcAndSetSceneRect(maxX, minX, maxY, minY, chartDataKind,

264 
265  // assign chartPoinOptions
266  chartPointOptions = &zv_termChartPointOptions;
267  }
268  else if(chartDataKind == CDK_CALIBRATION)
269  {

271  if(!zh_getCalibrationToConcentrationData(chartPointMap, maxX, minX, maxY, minY))
272  {

274  // return;

278  &zv_calibrationChartPointOptions);

280  // assign chartPoinOptions

282  }

285  zv_defaultItem->zp_setMedianaLineVisible(false);
286  if(!zh_getDeviationToConcentrationData(chartPointMap, maxX, minX, maxY, minY))

297  }
298  else
299  {
300  return;
301  }
302 
303  QMap<qint64, ZVisibilityPointF>::const_iterator it;
304  for(it = chartPointMap.begin(); it != chartPointMap.end(); it++)
305  {
306  ZChartPointGraphicsItem* pointItem;
307  bool itemExists = false;
308  for(int i = 0; i < chartPointItemList.count(); i++ )
309  {
310  pointItem = qgraphicsitem_cast<ZChartPointGraphicsItem*>(chartPointItemList.value(i));
311  if(!pointItem)
312  {
313  continue;
314  }
315 
316  if(pointItem->zp_relatedObjectId() == it.key())
317  {
318  // remove pointer from list (in order to increse performance)
319  chartPointItemList.removeAt(i);
320  itemExists = true;
321  break;
322  }
323  }
324 
325  // if item doesn't exist create it!
326  // if item exists, change it's pos and visibility
327 
328  if(itemExists)
329  {
330  pointItem->zp_applyVisibilityAndPos(it.value(), chartPointOptions);
331  }
332  else
333  {
334  pointItem = new ZChartPointGraphicsItem(this, it.value(), chartPointOptions, it.key());
335  zv_plotter->zp_addItem(pointItem);
336  }
337  }
338 
339  foreach(QGraphicsItem* item, chartPointItemList)
340  {
341  zv_plotter->zp_removeItem(item);
342  }
343 
344  if(!plotScaled)
345  {
346  zv_plotter->zp_fitInBoundingRect();
347 // QMetaObject::invokeMethod(zv_plotter, "zp_fitInBoundingRect",
348 // Qt::QueuedConnection);
349 
350  }
