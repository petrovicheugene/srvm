120 void ZAbstractTerm::zh_conformStringWithValue()
121 {
122  if(qAbs(zv_termFactor) < 1)
123  {
124  bool started = false;
125  zv_termFactorString = QString::number(zv_termFactor, 'f', zv_precision);
126  // first zero count after point
127  int firstZeroCount = 0;
128  for(int i = 0; i < zv_termFactorString.count(); i++)
129  {
130  if(!started && (zv_termFactorString.at(i) == '.' || zv_termFactorString.at(i) == ','))
131  {
132  started = true;
133  continue;
134  }
135 
136  if(zv_termFactorString.at(i) == '0')
137  {
138  firstZeroCount++;
139  continue;
140  }
141 
142  break;
143  }
144 
145  if(firstZeroCount > 9)
146  {
147  zv_termFactorString = QString::number(zv_termFactor, 'g', zv_precision);
148  }
149  else
150  {
151  // chop tail zero if they exist
152  for(int i = zv_termFactorString.count()-1; i > 0; i--)
153  {
154  if(zv_termFactorString.at(i) == '0' && (zv_termFactorString.at(i - 1) != '.' || zv_termFactorString.at(i - 1) != ','))
155  {
156  zv_termFactorString.remove(i, 1);
157  continue;
158  }
159  break;
160  }
161  }
162  }
163  else
164  {
165  zv_termFactorString = QString::number(zv_termFactor, 'g', zv_precision);
166  }
167 }
