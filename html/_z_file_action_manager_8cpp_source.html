262 {

264 
265 
266  // ioHandler subclass definition
267  // here is an ability to using the other classes, that handle files of different formats
268  if(!fileInfo.exists() || !fileInfo.isFile())

275  else if(fileInfo.suffix() != "xml" && fileInfo.suffix() != "spar")

278  QMessageBox::critical(0, tr("File handling error"), msg);

280  return false;
281  }
282 
283  // file opening
284  QFile file(fileName);
285  if(!file.open(QIODevice::ReadOnly))
286  {
287  QString errorMsg;
288  if(file.error() != QFile::NoError)

296  emit zg_message(errorMsg);
297  return false;

299

301
