428 bool ZSpectrumArray::zp_appendSpectrum(const ZRawSpectrum& rawSpectrum, bool last, bool& dontAsk)
429 {
430  QFileInfo fileInfo(rawSpectrum.path);
431  ZAbstractSpectrumIOHandler* ioHandler;
432 
433  QString suffix = fileInfo.suffix();
434  if(!fileInfo.exists() || !fileInfo.isFile())
435  {
436  QString error = tr("Error: \"%1\" is not a file!").arg(rawSpectrum.path);
437  emit zg_message(error);
438  return false;
439  }
440  else if(suffix == "spe")
441  {
442  ioHandler = new ZSpeIOHandler(this, this);
443  }
444  else
445  {
446  QString error = QObject::tr("Cannot handle file of type \"%1\"!").arg(suffix);
447  emit zg_message(error);
448  return false;
449  }
450 
451  ZAbstractSpectrum* abstractSpectrum;
452  bool res = ioHandler->zp_getSpectrumFromFile(rawSpectrum.path, zv_colorList.value(zv_lastColorIndex), abstractSpectrum);
453  if(res)
454  {
455  ZSpeSpectrum* speSpectrum = qobject_cast<ZSpeSpectrum*>(abstractSpectrum);
456 
457  if(zp_isEnergyCalibrationValid() && !dontAsk)
458  {
459  QStringList inconsistenciesList = speSpectrum->zp_isEnergyCalibrationSuitable(zv_energyUnit, zv_energyK0, zv_energyK1, zv_energyK2);
460  if(!inconsistenciesList.isEmpty())
461  {
462  // QString question = tr("Energy calibration in spectrum %1% is different to calibration that have other spectra in the array.\r\n").arg(rawSpectrum.path);
463  QString question = tr("The following inconsistencies have been found: \r\n");
464  for(int i = 0; i < inconsistenciesList.count(); i++)
465  {
466  question += "- "+inconsistenciesList.value(i);
467  if(i < inconsistenciesList.count() - 1)
468  {
469  question += ";";
470  }
471  else
472  {
473  question += ".";
474  }
475  question += "\r\n";
476  }
477  question += "Do you want to continue loading?";
478 
479  QMessageBox msgBox;
480  msgBox.setText(tr("Energy calibration in spectrum \"%1\" is different to calibration that have other spectra in the array.\r\n").arg(rawSpectrum.path));
481  msgBox.setInformativeText(question);
482 
483  if(last)
484  {
485  msgBox.setStandardButtons(QMessageBox::Yes | QMessageBox::No);
486  }
487  else
488  {
489  msgBox.setStandardButtons(QMessageBox::Yes | QMessageBox::No | QMessageBox::YesToAll | QMessageBox::NoToAll);
490  }
491  msgBox.setDefaultButton(QMessageBox::Yes );
492  int ret = msgBox.exec();
493 
494  switch (ret)
495  {
496  case QMessageBox::Yes:
497  res = true;
498  break;
499  case QMessageBox::YesToAll:
500  dontAsk = true;
501  res = true;
502  break;
503  case QMessageBox::No:
504  res = false;
505  break;
506  case QMessageBox::NoToAll:
507  default:
508  dontAsk = true;
509  res = false;
510  // should never be reached
511  break;
512  }
513  }
514  }
515 
516  if(res)
517  {
518  if(zv_spectrumList.count() < 1)
519  {
520  // first spectrum. storing energy calibration
521 
522  const ZSpeAuxData* auxData = static_cast<const ZSpeAuxData*>(speSpectrum->zp_auxData());
523  zv_energyUnit = auxData->zp_energyUnit();
524  zv_energyK0 = auxData->zp_energyK0();
525  zv_energyK1 = auxData->zp_energyK1();
526  zv_energyK2 = auxData->zp_energyK2();
527  emit zg_energyCalibrationChanged(zv_arrayId);
528  }
529 
530  int spectrumIndex = zv_spectrumList.count();
531  emit zg_spectrumOperation(OT_INSERT_SPECTRA, spectrumIndex, spectrumIndex);
532  zv_spectrumList.append(speSpectrum);
533  emit zg_spectrumOperation(OT_END_INSERT_SPECTRA, spectrumIndex, spectrumIndex);
534  // color index increment
535  if(++zv_lastColorIndex >= zv_colorList.count())
536  {
537  zv_lastColorIndex = 0;
538  }
539 
540  QMap<QString, QString>::const_iterator it;
541  qint64 chemElementId = -1;
542 
543  for(it = rawSpectrum.concentrationMap.begin(); it != rawSpectrum.concentrationMap.end(); it++)
544  {
545  if(!zv_chemElementList.zp_containsElement(it.key()))
546  {
547  zv_chemElementList.zp_appendElement(it.key(), chemElementId);
548  }
549  else
550  {
551  for(int i = 0; i < zv_chemElementList.zp_chemElementCount(); i++)
552  {
553  if(zv_chemElementList.zp_chemElementName(i) == it.key())
554  {
555  chemElementId = zp_chemElementId(i);
556  break;
557  }
558  }
559  }
560 
561  speSpectrum->zp_insertConcentration(chemElementId, it.value());
562  }
563  }
564  }
565 
566  delete ioHandler;
567  zh_recalcArrayMaxParameters();
568  return res;
569 }
