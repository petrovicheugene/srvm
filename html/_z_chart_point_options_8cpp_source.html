229 void ZChartPointOptions::zh_recalcShapeAndBoundingRect()
230 {
231  // shape
232  qreal pointHalfSize = (qreal)zv_pointSize / 2.0;
233  QRectF rect = QRectF(-1*pointHalfSize, -1*pointHalfSize,
234  (qreal)zv_pointSize, (qreal)zv_pointSize);
235 
236  QPainterPath newShape;
237  QPolygonF polygon;
238  qreal a;
239  switch(zv_pointType)
240  {
241  case PT_SQUARE:
242  newShape.addRect(rect);
243  break;
244  case PT_TRIANGLE:
245  a = pointHalfSize * 3.0 / sqrt(3.0);
246  polygon.append(QPointF(0.0, -1*pointHalfSize));
247  polygon.append(QPointF(-1*a/2, pointHalfSize/2));
248  polygon.append(QPointF(a/2, pointHalfSize/2));
249  polygon.append(QPointF(0.0, -1*pointHalfSize));
250 
251  newShape.addPolygon(polygon);
252  break;
253  case PT_ROUND:
254  newShape.addEllipse(rect);
255  break;
256  default:
257  newShape.addEllipse(rect);
258  }
259 
260  zv_shape = newShape;
261  zv_boundingRect = rect;
262 }
