664 {

667  {
668  zv_mouseButtonDown = true;
669  bool res = QObject::eventFilter(obj, event);
670  return res;
671  }else if(event->type() == QEvent::MouseButtonRelease

673  {
674  zv_mouseButtonDown = false;

676  zv_plotView->zp_setScaleViewWhenResizeFlag(false);

803 bool ZPlotter::zh_recalcVerticalDistortionFactors(qreal distortionValue)
804 {
805  if(zv_verticalAbsMax == 0)
806  {
807  return false;
808  }
809 
810  if(distortionValue < 0)
811  {
812  distortionValue = 0;
813  }
814  else if(distortionValue > 15)
815  {
816  distortionValue = 15;
817  }
818  zv_verticalDistortionFactor = 1.0 - (qreal)distortionValue / 20.0;
819  zv_verticalDistortionCorrectionFactor = zv_verticalAbsMax / pow(zv_verticalAbsMax, zv_verticalDistortionFactor);
820 
821  return true;
822 }
