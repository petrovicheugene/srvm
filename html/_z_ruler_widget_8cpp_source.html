571 void ZRulerWidget::zh_paintLeftRule(QPainter* painter)
572 {
573  if(!zv_leftRulerProperty.visibility || zv_YRuleList == 0)
574  {
575  return;
576  }
577  painter->save();
578 
579  QFont font = this->font();
580  font.setPixelSize(zv_leftRulerProperty.fontSize);
581  QFontMetrics fontMetrics(font);
582  QRect markRect;
583  painter->setFont(font);
584 
585  int y;
586  for(int i = 0; i < zv_YRuleList->count(); i++)
587  {
588  // scratch
589  y = mapFromGlobal(QPoint(0, qRound(zv_YRuleList->value(i).globalPos))).y();
590 
591  if(zv_leftRulerProperty.scratchRect.isValid())
592  {
593  painter->drawLine(zv_leftRulerProperty.scratchRect.left() , y,
594  zv_leftRulerProperty.scratchRect.right(), y);
595 
596  }
597  // Mark
598  if(zv_leftRulerProperty.markRect.isValid() && zv_YRuleList->value(i).markType != RulePoint::MT_SCRATCH)
599  {
600  // scratch
601  painter->drawLine(zv_leftRulerProperty.markRect.right() - zv_leftRulerProperty.scratchLineSize, y,
602  zv_leftRulerProperty.markRect.right(), y);
603 
604 
605  // mark
606  if(zv_YRuleList->value(i).markType == RulePoint::MT_MARK)
607  {
608  qreal markValue = zv_YRuleList->value(i).markValue * -1;
609  if(zv_leftRulerProperty.recalcMarkLabel)
610  {
611  markValue = pow(markValue, 2.0)*zv_leftRulerProperty.K2
612  + markValue*zv_leftRulerProperty.K1
613  + zv_leftRulerProperty.K0;
614  }
615  QString mark = QString::number(markValue, zv_leftRulerProperty.format, zv_leftRulerProperty.precision);
616  markRect = QRect(QPoint(0, 0), QSize(fontMetrics.width(mark), zv_maxMarkHeight));
617  markRect.moveCenter(QPoint(zv_leftRulerProperty.markRect.right()
618  - zv_leftRulerProperty.scratchLineSize
619  - zv_leftRulerProperty.markOffsetFromScratch
620  - markRect.width()/2, y));
621  painter->drawText(markRect, mark);
622  }
623  }
624  }
625 
626  if(zv_leftRulerProperty.ruleLabelVisibility)
627  {
628  painter->rotate(-90);
629  QRect ruleLabelRectForRotatedPainter;
630  ruleLabelRectForRotatedPainter.setBottom(zv_leftRulerProperty.ruleLabelRect.right());
631  ruleLabelRectForRotatedPainter.setTop(zv_leftRulerProperty.ruleLabelRect.left());
632  ruleLabelRectForRotatedPainter.setRight(zv_leftRulerProperty.ruleLabelRect.top() * -1);
633  ruleLabelRectForRotatedPainter.setLeft(zv_leftRulerProperty.ruleLabelRect.bottom() * -1);
634  painter->setRenderHint(QPainter::Antialiasing, true);
635  painter->drawText(ruleLabelRectForRotatedPainter, Qt::AlignCenter, zv_leftRulerProperty.ruleLabel);
636  }
637 
638  painter->restore();
639 }
