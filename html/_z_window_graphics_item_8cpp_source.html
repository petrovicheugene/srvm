178 void ZWindowGraphicsItem::zh_recalcShapeAndBoundingRect()
179 {
180  prepareGeometryChange();
181  // shape
182  QPainterPath newShape;
183  QPointF topLeftPoint(zv_leftMargin, zv_topMargin);
184  QPointF bottomRightPoint(zv_rightMargin, zv_bottomMargin);
185  QRectF windowRect(topLeftPoint, bottomRightPoint);
186 
187  newShape.addRect(windowRect);
188  zv_shape = newShape;
189  zv_boundingRect = windowRect;
190 
191  // paint rect shape
192  QPainterPath newRectPaintShape;
193  qreal shift = (qAbs(zv_bottomMargin - zv_topMargin)) / 2;
194  topLeftPoint = QPointF(zv_leftMargin, zv_topMargin - shift);
195  bottomRightPoint = QPointF(zv_rightMargin, zv_bottomMargin + shift);
196  QRectF paintRect(topLeftPoint, bottomRightPoint);
197  newRectPaintShape.addRect(paintRect);
198  zv_paintRect = newRectPaintShape;
199 
200  // line shape
201  QPainterPath newLinePaintShape;
202  QPointF bottomLeftPoint = QPointF(zv_leftMargin, zv_bottomMargin + shift);
203  QPointF topRightPoint = QPointF(zv_rightMargin, zv_topMargin - shift);
204 
205  newLinePaintShape.moveTo(topLeftPoint);
206  newLinePaintShape.lineTo(bottomLeftPoint);
207  newLinePaintShape.moveTo(topRightPoint);
208  newLinePaintShape.lineTo(bottomRightPoint);
209 
210  zv_marginsPainterPath = newLinePaintShape;
211 }
