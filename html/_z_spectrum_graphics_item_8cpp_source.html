122 void ZSpectrumGraphicsItem::zh_calculatePainterPath()
123 {
124  prepareGeometryChange();
125  QPointF point(0.0, 0.0);
126  QPainterPath newShape;
127  newShape.moveTo(point);
128 
129  qreal scValue = 0.0;
130  for(int i = 0; i < zv_spectrumData.count(); i++)
131  {
132  scValue = zh_sceneDistortionPositionForValue((qreal)(zv_spectrumData.value(i)));
133  point = QPointF((qreal)i, scValue*-1);
134  newShape.lineTo(point);
135  point = QPointF((qreal)(i+1), scValue*-1);
136  newShape.lineTo(point);
137  }
138 
139  // last point
140  point = QPointF((qreal)(zv_spectrumData.count() - 1), 0);
141  newShape.lineTo(point);
142  zv_spectrumPainterPath = newShape;
143  zv_boundingRect = zv_spectrumPainterPath.boundingRect();
144  // rect top and bottom extension
145  zv_boundingRect.setTop(zv_boundingRect.top()*zv_boundingRectTopFactor);
146  zv_boundingRect.setBottom(zv_boundingRect.bottom()+5);
147 }
