180 bool ZTermNormalizer::zh_calcNormaValue(const ZAbstractSpectrum *spectrum, qreal& normaValue) const
181 {
182  normaValue = 0;
183  if(!spectrum)
184  {
185  return false;
186  }
187 
188  if(zv_normaType == NT_NONE)
189  {
190  return true;
191  }
192 
193  if(zv_normaType == NT_COHERENT)
194  {
195  qint64 windowIntensity = 0;
196  if(!zh_getWindowsValue(ZCalibrationWindow::WT_COHERENT, spectrum, windowIntensity))
197  {
198  return false;
199  }
200  normaValue = static_cast<qreal>(windowIntensity);
201  }
202  else if(zv_normaType == NT_INCOHERENT)
203  {
204  qint64 windowIntensity = 0;
205  if(!zh_getWindowsValue(ZCalibrationWindow::WT_INCOHERENT, spectrum, windowIntensity))
206  {
207  return false;
208  }
209  normaValue = static_cast<qreal>(windowIntensity);
210  }
211  else if(zv_normaType == NT_COHERENT_INCOHERENT)
212  {
213  qint64 coherentValue = 0;
214  qint64 incoherentValue = 0;
215  if(!zh_getWindowsValue(ZCalibrationWindow::WT_INCOHERENT, spectrum, incoherentValue))
216  {
217  return false;
218  }
219 
220  if(!zh_getWindowsValue(ZCalibrationWindow::WT_COHERENT, spectrum, coherentValue))
221  {
222  return false;
223  }
224 
225  if(incoherentValue == 0)
226  {
227  return false;
228  }
229 
230  normaValue = static_cast<qreal>(coherentValue) / static_cast<qreal>(incoherentValue);
231  }
232  else if(zv_normaType == NT_INCOHERENT_COHERENT)
233  {
234  qint64 coherentValue = 0;
235  qint64 incoherentValue = 0;

245

250
